## 说明

用于触发操作，不同样式表达不同层级。

## 用法

```tsx
import { Button } from '@/components/ui/button';

export function Example() {
  return (
    <>
      <Button>主要</Button>
      <Button variant="secondary">次要</Button>
      <Button variant="outline">描边</Button>
      <Button variant="ghost">幽灵</Button>
      <Button variant="destructive">危险</Button>
    </>
  );
}
```

## 说明

用于自定义滚动容器。

```tsx
import { ScrollArea } from "@/components/ui/scroll-area"

export function Example() {
  return (
    <ScrollArea className="h-40 rounded-md border p-4">
      <div className="space-y-2">
        <div>Item 1</div>
        <div>Item 2</div>
        <div>Item 3</div>
      </div>
    </ScrollArea>
  )
}
```

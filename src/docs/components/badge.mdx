## 说明

用于展示状态或分类标签。

## 用法

```tsx
import { Badge } from '@/components/ui/badge';

export function Example() {
  return (
    <>
      <Badge>默认</Badge>
      <Badge variant="secondary">次要</Badge>
      <Badge variant="destructive">警告</Badge>
      <Badge variant="outline">描边</Badge>
    </>
  );
}
```

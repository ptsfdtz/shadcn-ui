## 说明

用于展示列表项或卡片条目。

```tsx
import {
  Item,
  ItemContent,
  ItemDescription,
  ItemGroup,
  ItemTitle,
} from "@/components/ui/item"

export function Example() {
  return (
    <ItemGroup>
      <Item>
        <ItemContent>
          <ItemTitle>Notifications</ItemTitle>
          <ItemDescription>Manage alerts and reminders.</ItemDescription>
        </ItemContent>
      </Item>
    </ItemGroup>
  )
}
```
